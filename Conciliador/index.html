function onFormSubmit(e) { // Obtém as respostas do formulário var responses = e.values;

// Substitua 'Sua Chave de API Asana' pela sua chave de API Asana var apiKey = 'Sua Chave de API Asana';

// Substitua 'ID do Projeto Asana' pelo ID do seu projeto Asana var projectId = 'ID do Projeto Asana';

// Substitua 'Nome da Tarefa Asana' pelo nome que você deseja dar à tarefa no Asana var taskName = 'Nome da Tarefa Asana';

// Substitua 'ID da Seção Asana' pelo ID da seção do seu projeto Asana (opcional) var sectionId = 'ID da Seção Asana';

// Cria a tarefa no Asana var taskId = createAsanaTask(apiKey, projectId, taskName, sectionId);

// Adiciona um comentário na tarefa do Asana com as respostas do formulário addCommentToAsanaTask(apiKey, taskId, responses); }

function createAsanaTask(apiKey, projectId, taskName, sectionId) { var apiUrl = 'https://app.asana.com/api/1.0/tasks'; var headers = { 'Authorization': 'Bearer ' + apiKey, 'Content-Type': 'application/json', };

var payload = { data: { projects: [projectId], name: taskName, }, };

if (sectionId) { payload.data.section = sectionId; }

var options = { method: 'post', headers: headers, payload: JSON.stringify(payload), };

var response = UrlFetchApp.fetch(apiUrl, options); var responseData = JSON.parse(response.getContentText());

return responseData.data.id; }

function addCommentToAsanaTask(apiKey, taskId, comments) { var apiUrl = 'https://app.asana.com/api/1.0/tasks/' + taskId + '/stories'; var headers = { 'Authorization': 'Bearer ' + apiKey, 'Content-Type': 'application/json', };

var payload = { data: { text: 'Respostas do Formulário:\n\n' + comments.join('\n'), }, };

var options = { method: 'post', headers: headers, payload: JSON.stringify(payload), };

UrlFetchApp.fetch(apiUrl, options); }
